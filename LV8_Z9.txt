#include <stdio.h>
#include <math.h>
struct Tacka{
    double x,y;
};
struct Kruznica{
    struct Tacka centar;
    double poluprecnik;
};
double daj_duzinu(struct Tacka A, struct Tacka B){
    return sqrt((double)(A.x-B.x)*(A.x-B.x)+(A.y-B.y)*(A.y-B.y));
}
struct Kruznica obuhvat(struct Tacka*niz, int vel){
    struct Kruznica k;
    double maxX=niz[0].x,maxY=niz[0].y,minX=niz[0].x,minY=niz[0].y;
    int i,j;
    
    k.poluprecnik=0;
    
    for(i=1;i<vel;i++){
        if(niz[i].x>maxX)maxX=niz[i].x;
        if(niz[i].y>maxY)maxY=niz[i].y;
        if(niz[i].x<minX)minX=niz[i].x;
        if(niz[i].y<minY)minY=niz[i].y;
    }
    k.centar.x=(maxX+minX)/2.0;
    k.centar.y=(maxY+minY)/2.0;
    double max=0;
    for(i=0;i<vel;i++){
        if(daj_duzinu(niz[i], k.centar)>max){
            max=daj_duzinu(niz[i], k.centar);
            k.poluprecnik=max;
        }
        
    }
    return k;
}
void ispis_kruznice(struct Kruznica k){
    printf("(%.2lf,%.2lf) %.2lf",k.centar.x,k.centar.y,k.poluprecnik);
}
int main() {
    struct Tacka niz[1];
    niz[0].x=45;
    niz[0].y=45;
    struct Kruznica k=obuhvat(niz, 1);
    ispis_kruznice(k);
	return 0;
}
