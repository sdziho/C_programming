
#include <stdio.h>

int main() {
	int n,i,j,m;
    printf("Unesite broj n: ");
    scanf("%d", &n);
    if(n<4 || n%2!=0 || n>100){
        printf("Broj je van trazenog opsega.");
        return 0;
    }
    m=2*n;
    for(i=0;i<n;i++){
        for(j=0;j<m;j++){
            if(i==0||j==0||i==n-1||j==m-1) printf("+");
            else if(i==j && i<n/2-1) printf("+");
            else if(i==n/2-1 &&j>=n/2-1 &&j<m-n/2+1) printf("+");
            else if(m-1-j==i && i<n/2-1) printf("+");
            else printf(" ");
        }
        printf("\n");
    }
	return 0;
}
