#include <stdio.h>
int daj_sljedeci_broj(int param){
    static int dojnja=0;
    static int gornja=100;
    if(param==0) return dojnja+(gornja-dojnja)/2;
    if(param==1){
        dojnja=dojnja+(gornja-dojnja)/2;
        return dojnja+(gornja-dojnja)/2;
    }
    if(param==-1){
        gornja=dojnja+(gornja-dojnja)/2;
        return dojnja+(gornja-dojnja)/2;
    }
    return 0;
}
int main() {
    int n;
    char c;
    n=50;
	printf("Zamislite neki broj...\n");
    while(1){
        printf("Da li je taj broj %d? ",daj_sljedeci_broj(0));
        scanf(" %c",&c);
        if(c=='M')daj_sljedeci_broj(-1);
        else if(c=='V')daj_sljedeci_broj(1);
        else if(c=='J') break;
        else printf("Pogresan unos, probajte ponovo.\n");
    }
    printf("Pogodio sam!");
	return 0;
}
