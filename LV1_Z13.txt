#include <stdio.h>

int main() {
	int n,counter,broj;
    int n1,n2,n3,n4,n5,n6; //maksimalan broj cifara je 6 pa imamo 6 varijablih
    printf("Unesite broj: ");
    scanf("%d",&n);
    if(n<=-1000000||n>=1000000){
        printf("Neispravan unos!");
        return 0;
    }
    printf("Koju cifru zelite kvadrirati: ");
    scanf("%d",&broj);
    if(n<0) n=n*(-1);
    counter=0;
    n1=n%10;//5
    n2=(n/10)%10;//1
    n3=(n/100)%10;//5
    n4=(n/1000)%10;//4
    n5=(n/10000)%10;//2
    n6=(n/100000)%10;
    
    if(n!=0) counter++;
    if(n/10!=0) counter++;
    if(n/100!=0) counter++;
    if(n/1000!=0) counter++;
    if(n/10000!=0) counter++;
    if(n/100000!=0) counter++;
    //counter=5
    if(counter<broj || broj<1) {
        printf("Neispravna cifra!");
        return 0;
    }
    if(counter-broj==0){
        printf("Izdvojena cifra je %d, a njen kvadrat je %d",n1,n1*n1);
        return 0;
    }
    if(counter-broj==1){
        printf("Izdvojena cifra je %d, a njen kvadrat je %d",n2,n2*n2);
        return 0;
    }
    if(counter-broj==2){
        printf("Izdvojena cifra je %d, a njen kvadrat je %d",n3,n3*n3);
        return 0;
    }
    if(counter-broj==3){
        printf("Izdvojena cifra je %d, a njen kvadrat je %d",n4,n4*n4);
        return 0;
    }
    if(counter-broj==4){
        printf("Izdvojena cifra je %d, a njen kvadrat je %d",n5,n5*n5);
        return 0;
    }
    if(counter-broj==5){
        printf("Izdvojena cifra je %d, a njen kvadrat je %d",n6,n6*n6);
        return 0;
    }

	return 0;
}
