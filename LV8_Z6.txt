#include <stdio.h>
struct Student{
    char ime[20],prezime[20];
    int ocjene[50];
    int br_ocjena;
};

int genijalci(struct Student *s,int vel, double prosjek){
    int i,j,counter,flag,k,z;
    double suma;
    int max1=-1,max2=-1,max3=-1;
    double maxS=0;
    for(i=0;i<vel;i=i+1){
        counter=0;
        suma=0;
        flag=0;
        for(j=0;j<s[i].br_ocjena;j++){
            if(s[i].ocjene[j]<6){ flag=0;break;}
            suma+=s[i].ocjene[j];
            counter++;
            flag=1;
        }
        if(flag==0) suma=5.0;
        else {
            suma/=counter;
            
        }
        if(suma<prosjek){
            for(j=i;j<vel-1;j++)s[j]=s[j+1];
            vel--;
            i--;
        }

        if(suma>maxS&&suma!=5.0){
                maxS=suma;
                max1=i;
            }
        else if(suma==maxS){
            k=0;
            while(s[i].prezime[k]==s[max1].prezime[k]){
                k++;
                if(s[max1].prezime[k]=='\0'){
                    z=0;
                    while(s[i].ime[z]==s[max1].ime[z]||s[max1].ime[z]=='\0')z++;
                    if(s[i].ime[z]<s[max1].ime[z]) max1=i;
                    break;
                }
            }
            if(s[i].prezime[k]<s[max1].prezime[k]) max1=i;
            
        }
    }
    
    maxS=0;
    for(i=0;i<vel;i=i+1){
        counter=0;
        suma=0;
        flag=0;
        for(j=0;j<s[i].br_ocjena;j++){
            if(s[i].ocjene[j]<6){ flag=0;break;}
            suma+=s[i].ocjene[j];
            counter++;
            flag=1;
        }
        if(flag==0) suma=5.0;
        else {
            suma/=counter;
            if(suma>maxS&&i!=max1){
                maxS=suma;
                max2=i;
            }
            else if(suma==maxS&&i!=max1){
                k=0;
                while(s[i].prezime[k]==s[max2].prezime[k]){
                    k++;
                    if(s[max2].prezime[k]=='\0'){
                        z=0;
                        while(s[i].ime[z]==s[max2].ime[z]||s[max2].ime[z]=='\0')z++;
                        if(s[i].ime[z]<s[max2].ime[z]) max2=i;
                        break;
                    }
                }
                if(s[i].prezime[k]<s[max2].prezime[k]) max2=i;
                
            }
        }
    }

    maxS=0;
    for(i=0;i<vel;i=i+1){
        counter=0;
        suma=0;
        flag=0;
        for(j=0;j<s[i].br_ocjena;j++){
            if(s[i].ocjene[j]<6){ flag=0;break;}
            suma+=s[i].ocjene[j];
            counter++;
            flag=1;
        }
        if(flag==0) suma=5.0;
        else {
            suma/=counter;
            if(suma>maxS&&i!=max1&&max2!=i){
                maxS=suma;
                max3=i;
            }
            else if(suma==maxS&&i!=max1&&max2!=i){
                k=0;
                while(s[i].prezime[k]==s[max3].prezime[k]){
                    k++;
                    if(s[max3].prezime[k]=='\0'){
                        z=0;
                        while(s[i].ime[z]==s[max3].ime[z]||s[max3].ime[z]=='\0')z++;
                        if(s[i].ime[z]<s[max3].ime[z]) max3=i;
                        break;
                    }
                }
                if(s[i].prezime[k]<s[max3].prezime[k]) max3=i;
                
            }
        }
    }
    if(max1!=-1)printf("%s %s\n",s[max1].prezime,s[max1].ime);
    if(max2!=-1)printf("%s %s\n",s[max2].prezime,s[max2].ime);
    if(max3!=-1)printf("%s %s\n",s[max3].prezime,s[max3].ime);
    return vel;
}
int main() {
	return 0;
}
