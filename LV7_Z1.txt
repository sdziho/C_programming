#include <stdio.h>

int obrnut_podstring(const char *s1, const char *s2){
    int i,flag;
    const char*p=s1;
    const char*q=s2;
    while(*q!='\0')q++;
    while(*p!='\0')p++;
    if((p-s1)<(q-s2)) return 0;
    q--;
    
    while(*s1!='\0'){
        if(*s1==*q){
            flag=1;
            for(i=1;i<q-s2+1;i++){
                if(*(s1+i)!=*(q-i))flag=0;
            }
            if(flag==1) return 1;
        }
        s1++;
    }
    return 0;
}
int main() {
	if(obrnut_podstring("a", "aa")) printf("DA");
    else printf("NE");
	return 0;
}
