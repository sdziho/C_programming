#include <stdio.h>

int main() {
	int n,m,broj,i,j;
    printf("Unesite broj n: ");
    scanf("%d", &broj);
    if(broj%2==0 || broj<2 || broj>100){
        printf("Broj je van trazenog opsega");
        return 0;
    }
    n=broj+broj/2;
    m=broj;
    for(i=0;i<n;i++){
        for(j=0;j<m;j++){
            if(i==broj/2 || i==n-1) printf("+");
            else if((j==0||j==m-1)&&i>=broj/2) printf("+");
            else if(i<broj/2&&j==broj/2+i) printf("+");
            else if(i<broj/2&&j==broj/2-i) printf("+");
            else if(i-broj/2==j) printf("+");
            else if(i==broj-j+broj/2-1) printf("+");
            else printf(" ");

        }
        printf("\n");
    }
	return 0;
}
