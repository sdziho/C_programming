#include <stdio.h>
int fibonacci(int n){
    int zadnja,prezadnja,i,temp;
    zadnja=1;
    prezadnja=1;
    if(n==0) return 0;
    for(i=2;i<n;i++){
        temp=zadnja;
        zadnja=zadnja+prezadnja;
        prezadnja=temp;
    }
    return zadnja;
}
int main() {
	printf("%d",fibonacci(5));
	return 0;
}
