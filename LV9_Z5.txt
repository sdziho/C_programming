#include <stdio.h>

void unesi(char niz[], int velicina){
    char znak=getchar();
    if(znak=='\n') znak=getchar();
    int i=0;
    while(i<velicina-1&&znak!='\n'){
        niz[i]=znak;
        i++;
        znak=getchar();
    }
    niz[i]='\0';
}
int main() {
    char ulazna[20];
    int histogram[150]={0};
    int i,c;
	printf("Unesite naziv ulazne datoteke: ");
    unesi(ulazna, 20);
    FILE*ulaz=fopen(ulazna,"r");
    FILE*izlaz=fopen("statistika.txt","w");
    do{
        c=getc(ulaz);
        if(c==EOF)break;
        if(c>='a'&&c<='z') histogram[c-('a'-'A')]++;//64 do 91
        else if(c>='A'&&c<='Z') histogram[c]++;
    }while(c!=EOF);
    while(1){
        int max=0;
        int flag=0;
        for(i=63;i<92;i++){
            if(histogram[i]>0&&histogram[i]>histogram[max]) {
                max=i;
                flag=1;
            }
        }
        if(flag==0) break;
        histogram[max]=0;
        fputc(max,izlaz);
    }
    fclose(ulaz);
    fclose(izlaz);
	return 0;
}
