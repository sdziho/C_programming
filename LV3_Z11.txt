#include <stdio.h>
#include <math.h>
#define epsilon 0.0001
int main() {
	double niz[200],art,temp;
    int histo[10]={0};
    int n,i,j,cifra,pretho;
    do{
        printf("Unesite broj clanova niza: ");
        scanf("%d",&n);
    }while(n<=0||n>100);
    printf("Unesite niz: ");
    for(i=0;i<n;i++) scanf("%lf", &niz[i]);

    for(i=0;i<n-1;i++){
        if((2*niz[i]-niz[i+1])<epsilon && (2*niz[i]-niz[i+1])>-epsilon){
            art=(niz[i]+niz[i+1])/2;
            n++;
            for(j=n-1;j>i;j--) niz[j]=niz[j-1];
            niz[i+1]=art;
            i++;
        }
    }
    printf("Nakon ubacivanja niz glasi:\n");
    for(i=0;i<n;i++) printf("%g ", niz[i]);
    for(i=0;i<n;i++){
        temp=niz[i];
        temp=temp*10;
        cifra=(int)temp;
        histo[cifra%10]++;
    }
    for(i=0;i<10;i++) {
        if(histo[i]>0) printf("\nCifra %d se ponavlja %d puta.",i, histo[i]);
    }
	return 0;
}
