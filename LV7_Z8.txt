#include <stdio.h>

char* popuni_lijevo(char* str, int n){
    char*p=str,*kraj=str;
    while(*kraj!='\0')kraj++;
    if(n<(kraj-str)) return str;
    int razlika=n-(kraj-str),i;
    do{
        *(kraj+razlika)=*kraj;
        kraj--;
    }while(kraj>=str);
    for(i=0;i<razlika;i++)*(str+i)=' ';
    return str;
}
