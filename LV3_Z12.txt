#include <stdio.h>

int main() {
    int n,i,dani[1000],mjeseci[1000],godine[1000],j;
    int days[1000],d,m,g,prestupna,suma;
    do{
        printf("Unesite broj studenata: ");
        scanf("%d",&n);
        if(n<=0||n>1000) printf("Neispravan broj studenata\n");
    }while(n<=0||n>1000);
    for(i=0;i<n;i=i+1){
        printf("Unesite datum rodjenja %d. studenta: ", i+1);
        scanf("%d,%d,%d",&d,&m,&g);
        prestupna=0;
        if(m==2 && g%4==0) prestupna=1;
        if(g%100==0 &&g%400!=0) prestupna=0;
        if(m>12||m<=0 || d<=0||g<=0) printf("Neispravan datum \n");
        else if((m==1||m==3||m==5||m==7||m==8||m==10||m==12) && d>31) printf("Neispravan datum\n"); 
        else if((m==4||m==6||m==9||m==11) && d>30) printf("Neispravan datum\n"); 
        else if((m==2 && d>28 && prestupna==0) || (m==2 && d>29 &&prestupna==1) ) printf("Neispravan datum\n"); 
        else {
            dani[i]=d;
            godine[i]=g;
            mjeseci[i]=m;
            continue;
        }
        i--;
    }
    for(i=0;i<n;i++){
        suma=0;
        for(j=1;j<godine[i];j++){
            g=j;
            prestupna=0;
            if(g%4==0) prestupna=1;
            if(g%100==0 &&g%400!=0) prestupna=0;
            if(prestupna==1) suma+=366;
            else suma+=365;
        }
        m=mjeseci[i];
        if(m>1) suma+=31;
        if(m>2) suma+=28;
        if(m>3) suma+=31;
        if(m>4) suma+=30;
        if(m>5) suma+=31;
        if(m>6) suma+=30;
        if(m>7) suma+=31;
        if(m>8) suma+=31;
        if(m>9) suma+=30;
        if(m>10) suma+=31;
        if(m>11) suma+=30;
        if(m>12) suma+=31;

        g=godine[i];
        if(g%4==0&&m>2) prestupna=1;
        if(g%100==0 &&g%400!=0) prestupna=0;
        if(prestupna==1) suma++;

        suma+=dani[i];
        days[i]=suma;
        //printf("%d\n",suma);
    }

    for(i=0;i<n-1;i++){
        for(j=i;j<n;j++){
            if(days[i]-days[j]==100 || days[j]-days[i]==100){
                if(days[i]<days[j])
                    printf("Student %d. je rodjen tacno 100 dana prije studenta %d.\n",i+1,j+1);
                else
                    printf("Student %d. je rodjen tacno 100 dana prije studenta %d.\n",j+1,i+1);
                
            }
        }
    }
    //printf("Student %d. je rodjen tacno 100 dana prije studenta %d.",i+1,j+1);
                
	return 0;
}
