#include <stdio.h>

int izbaci_cifre(int* niz,int n, int* cifre, int vel){
    int i,j,temp,desetice,suma;
    for(i=0;i<vel;i++){
        if(*(cifre+i)<0||*(cifre+i)>9) return 0;
        for(j=i+1;j<vel;j++){
            if(*(cifre+i)==*(cifre+j)) return 0;
        }
    }
    for(i=0;i<n;i++){
        for(j=0;j<vel;j++){
            desetice=1;
            temp=*(niz+i);
            if(temp<0)temp*=(-1);
            suma=0;
            while(temp>0){
                if(*(cifre+j)!=temp%10){
                    suma+=(temp%10)*desetice;
                    desetice*=10;
                }
                temp/=10;
            }
            if(*(niz+i)<0) suma*=(-1);
            *(niz+i)=suma;
        }
    }
    return 1;
}
int main() {
    int niz[]={123};
    int cifre[]={1};
    printf("%d ", izbaci_cifre(niz,1,cifre,1));
    printf("%d",niz[0]);
    
	return 0;
}
