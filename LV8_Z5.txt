#include <stdio.h>
#include <math.h>

struct Tacka{
    double koordinata[100];
};

int izbaci_najblize(struct Tacka *t, int vel, int dimenzija){
    int i,j,index1=0,index2=0,k,flag;
    double suma,dMax=9999;
    if(vel==1) return 1;
    for(i=0;i<vel-1;i++){
        for(j=i+1;j<vel;j++){
            suma=0;
            for(k=0;k<dimenzija;k++){
                suma+=pow(t[i].koordinata[k]-t[j].koordinata[k],2);
            }
            suma=sqrt(suma);
            
            if(suma<dMax){
                dMax=suma;
                index1=i;
                index2=j;
            }
        }
    }
    flag=0;
    for(i=0;i<vel;i++){
        if(i==index1&&flag==0){
            for(j=i;j<vel;j++)t[j]=t[j+1];
            vel--;
            i--;
            flag=1;
        }
        else if(flag==1&&i==index2-1){
            for(j=i;j<vel;j++)t[j]=t[j+1];
            vel--;
            break;
        }
    }
    return vel;
}
int main() {
    struct Tacka t[]={
        {{1,2}}
    };
    printf("%d",izbaci_najblize(t, 1, 2));
	return 0;
}
