#include <stdio.h>

int pronadji_posljednji(const char*str,const char*trazi){
    int ponavljanje=-1;
    const char*s1=str;
    const char*kraj1=str,*kraj2=trazi;
    while(*kraj1!='\0')kraj1++;
    while(*kraj2!='\0')kraj2++;
    if((kraj2-trazi)>(kraj1-str)) return ponavljanje;
    while(*str!='\0'){
        const char *p=str;
        const char *q=trazi;
        while(*p++==*q++)
            if(*q=='\0') {ponavljanje=str-s1;break;}
        str++;
    }
    return ponavljanje;
}
int main() {
	printf("%d", pronadji_posljednji("asa","asa"));
