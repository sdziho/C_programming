#include <stdio.h>

char *izbaci_maxbroj(char *s){
    char *pom=s;
    int max=123,prvi_put=1,i;
    while(*pom!='\0'){//mjenjaj
        if(*pom>='0'&&*pom<='9'){
            int minus=0,suma=0;
            if(pom>s){
                if(*(pom-1)=='-')minus=1;
            }
            while(*pom>='0'&&*pom<='9'){
                suma*=10;
                suma+=*pom-'0';
                pom++;
            }
            if(minus==1)suma*=(-1);
            if(prvi_put==1){
                max=suma;
                prvi_put=0;
                if(*pom=='\0')break;
                pom++;
                continue;
            }
            if(prvi_put==0&&suma>max)max=suma;
            continue;
        }
        pom++;
    }
    pom=s;
    while(*pom!='\0'){
        if(*pom=='-'||(*pom>='0'&&*pom<='9')){
            char *kraj=pom;
            int temp=max,minus=0;
            while(*kraj=='-'||(*kraj>='0'&&*kraj<='9'))kraj++;
            if(temp>0&&*pom=='-'){
                pom=kraj;
                continue;
            }
            if(temp<0){
                minus=1;
                temp*=(-1);
            }
            int flag=1;
            int br_cifara=0;
            int pomocni=temp;
            if(pomocni==0)br_cifara++;
            while(pomocni>0){
                br_cifara++;
                pomocni/=10;
            }
            for(i=0;i<br_cifara;i++){
                if(*(kraj-i-1)-'0'!=temp%10){
                    if(*(kraj-i-1)=='-'&&minus==1)flag=1;
                    else flag=0;
                    break;
                }
                temp/=10;
            }
            if(flag==1&&kraj-pom>0){
                char *p=pom;
                while(*p!='\0'){
                    *p=*(p+(kraj-pom));
                    if(*p=='\0')break;
                    p++;
                }
                
            }
        }
        if(*pom=='\0')break;
        pom++;
    }
    return s;
}
int main() {
    char niz[]="broj 1 a-700 100 i 010 su veci od 0";
    printf("%s",izbaci_maxbroj(niz));
	return 0;
}
