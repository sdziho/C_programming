#include <stdio.h>

int main() {
    double niz[1000],broj;
    int histogram[1000]={0};
    int i,n,max_index,j;
	printf("Unesite broj clanova niza: ");
    scanf("%d",&n);
    printf("Unesite clanove niza: ");
    for(i=0;i<n;i++) scanf("%lf",&niz[i]);
    max_index=0;
    for(i=0;i<n;i++){
        for(j=0;j<n;j++){
            if(i==j) continue;
            if(niz[i]==niz[j]) histogram[i]+=1;
        }
    }
    for(i=1;i<n;i++){
        //printf("%d ",histogram[i]);
        if(histogram[i]>histogram[max_index]) max_index=i;
        else if(histogram[i]==histogram[max_index]){
            if(niz[i]<niz[max_index])max_index=i;
        }
    }
    
    broj=niz[max_index];
    for(i=0;i<n;i=i+1){
        if(niz[i]>broj){
            for(j=i;j<n-1;j++) niz[j]=niz[j+1];
            n--;
            i=-1;
        }
    }
    printf("Nakon izbacivanja niz glasi:\n");
    for(i=0;i<n;i++) printf("%g ",niz[i]); //ispis izbacenih clanova, n je nova velicina niza

	return 0;
}
