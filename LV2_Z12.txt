#include <stdio.h>

int main() {
	int n, m;
    int i,j;
    printf("Unesite n: ");
    scanf("%d", &n);
    if(n<2){
        printf("Neispravan unos");
        return 0;
    }
    m=2*n;
    for(i=1;i<m;i++){
        for(j=0;j<m+1;j++){
            if(j==m/2) printf(" ");
            else if(i>j && i<=n) printf("*");
            else if (i<m-j && i>n) printf("*");
            else if (i>m-j && i<=n) printf("*");
            else if(i<j && i>n) printf("*");
            else printf(" ");
        }
        printf("\n");
    }

	return 0;
}
