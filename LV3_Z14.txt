#include <stdio.h>

int main() {
    int niz[100],n,cifre[1000],i,j,last,temp,counter,broj,flag;
	printf("Unesite velicinu niza: ");
    scanf("%d", &n);
    printf("Unesite clanove niza: ");
    for(i=0;i<n;i++)scanf("%d",&niz[i]);
    broj=0;
    last=niz[n-1]%10;
    counter=0;
    flag=1;
    for(i=n-1;i>=0;i--){
        temp=niz[i];
        if(temp==0){
            if(last==0) counter++;
            else{
                cifre[broj]=counter;
                cifre[broj+1]=last;
                broj+=2;
                last=0;
                counter=1;
            }
            continue; 
        }
        while(temp>0){
            if(last==temp%10) counter++;
            else{
                cifre[broj]=counter;
                cifre[broj+1]=last;
                broj+=2;
                counter=1;
            }
            last=temp%10;
            temp=temp/10;
        }

    }
    cifre[broj]=counter;
    cifre[broj+1]=last;
    broj+=2;
    for(i=0;i<broj/2;i++){
        temp=cifre[i];
        cifre[i]=cifre[broj-i-1];
        cifre[broj-i-1]=temp;
    } 
    printf("Finalni niz glasi:\n");
    for(i=0;i<broj;i++)printf("%d ", cifre[i]);
	return 0;
}
