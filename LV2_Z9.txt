#include <stdio.h>

int main() {
	int iznos=93,i;
    int n50=0,n20=0,n10=0,n5=0,n2=0,n1=0;
    printf("Unesite iznos: ");
    scanf("%d", &iznos);
    i=iznos;
    while(i>0){
        if(i-50>=0) {
            n50++;
            i=i-50;
            continue;
        }
        if(i-20>=0) {
            n20++;
            i=i-20;
            continue;
        }
        if(i-10>=0) {
            n10++;
            i=i-10;
            continue;
        }
        if(i-5>=0) {
            n5++;
            i=i-5;
            continue;
        }
        if(i-2>=0) {
            n2++;
            i=i-2;
            continue;
        }
        if(i-1>=0) {
            n1++;
            i=i-1;
            continue;
        }
    }
    printf("%d novcanica od 50 KM\n",n50);
    printf("%d novcanica od 20 KM\n",n20);
    printf("%d novcanica od 10 KM\n",n10);
    printf("%d novcanica od 5 KM\n",n5);
    printf("%d novcanica od 2 KM\n",n2);
    printf("%d novcanica od 1 KM\n",n1);
	return 0;
}
