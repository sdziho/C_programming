#include <stdio.h>

int main() {
    int n,i,j;
    double mat[30][30],O1,O2,O3,O4;
	printf("Unesite broj n: ");
    scanf("%d", &n);
    if(n<3||n>30){
        printf("Unos pogresan!");
        return 0;
    }
    for(i=0;i<n;i++){
        printf("Unesite elemente %d. reda: ",i+1);
        for(j=0;j<n;j++) scanf("%lf", &mat[i][j]);
    }
    O2=mat[1][0];
    O1=mat[0][1];
    O3=mat[1][n-1];
    O4=mat[n-1][n-2];
    for(i=0;i<n;i++){
        for(j=0;j<n;j++){
            if(i>j&&i<n-j-1){
                if(mat[i][j]<O2) O2=mat[i][j];
            }
            if(i>j&&i>n-j-1){
                if(mat[i][j]>O4) O4=mat[i][j];
            }
            if(i<j&&i<n-j-1){
                if(mat[i][j]>O1) O1=mat[i][j];
            }
            if(i<j&&i>n-j-1){
                if(mat[i][j]<O3) O3=mat[i][j];
            }
        }
    }
    if(O1==O4&&O2==O3) printf("Matrica je simetricna!");
    else printf("Matrica nije simetricna!");
    printf("\nElementi: O1 = %.3f, O2 = %.3f, O3 = %.3f, O4 = %.3f",O1,O2,O3,O4);
	return 0;
}
