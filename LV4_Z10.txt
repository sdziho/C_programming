#include <stdio.h>

int main() {
    double mat[100][100],suma;
    int n,m,i,j;
	printf("Unesite dimenzije matrice: ");
    scanf("%d %d",&n,&m);
    printf("Unesite elemente matrice: ");
    for(i=0;i<n;i++)
        for(j=0;j<m;j++)scanf("%lf",&mat[i][j]);

    suma=0;
    for(i=0;i<n;i++){
        for(j=0;j<m;j++){
            if(i==0||i==n-1||j==0||j==m-1) suma+=mat[i][j];
            
        }
    }
    printf("Suma elemenata na rubu je %.2f", suma);
	return 0;
}
