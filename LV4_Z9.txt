#include <stdio.h>

int main() {
    int n1,m1,A[10][10],n2,m2,B[10][10],C[10][10];
    int i, j,n3,m3;
    printf("Unesite dimenzije matrice A (p q): ");
    scanf("%d %d", &n1,&m1);
    if(n1<=0||m1<=0){
        printf("Neispravne dimenzije matrice!");
        return 0;
    }
    if(n1>10||m1>10){
        printf("Matrica A je prevelika!");
        return 0;
    }
    printf("Unesite dimenzije matrice B (r s): ");
    scanf("%d %d", &n2,&m2);
    if(n2<=0||m2<=0){
        printf("Neispravne dimenzije matrice!");
        return 0;
    }
    if(n2>10||m2>10){
        printf("Matrica B je prevelika!");
        return 0;
    }
    if(m2!=m1) {
        printf("Matrice nisu saglasne za konkatenaciju!");
        return 0;
    }
    printf("Matrica A: ");
    for(i=0;i<n1;i++){
        printf("%d. red: ",i+1);
        for(j=0;j<m1;j++) scanf("%d",&A[i][j]);
    }
    printf("Matrica B: ");
    for(i=0;i<n2;i++){
        printf("%d. red: ",i+1);
        for(j=0;j<m2;j++) scanf("%d",&B[i][j]);
    }
    for(i=0;i<n1+n2;i++){
        for(j=0;j<m1;j++){
            if(i<n1) C[i][j]=A[i][j];
            else C[i][j]=B[i-n1][j];
        }
    }
    n3=n1+n2;
    m3=m2;
    printf("Matrica C:\n");
    for(i=0;i<n3;i++){
        for(j=0;j<m3;j++) printf("%4d",C[i][j]);
        printf("\n");
    }
	return 0;
}
